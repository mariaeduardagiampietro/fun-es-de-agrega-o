SELECT SUM(ESTOQUE) AS TOTAL_ESTOQUE
FROM MEDICAMENTO

/* Atividade 2 */

SELECT AVG(EXTRACT(YEAR FROM CURRENT_DATE) - EXTRACT(YEAR FROM t.data_nascimento)) AS media_idade
FROM TUTOR t

/* Atividade 3 */

SELECT MIN (UNIDADE)
FROM MEDICAMENTOS

/* Atividade 4 */

SELECT MAX (PESO)
FROM PACIENTE

/* Atividade 5 */

SELECT COUNT (*)
FROM TUTOR

/* Atividade 6 */

SELECT SUM(ESTOQUE * PRECO)
FROM MEDICAMENTOS
GROUP BY UNIDADE
